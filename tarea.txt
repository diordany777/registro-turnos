let registros = JSON.parse(localStorage.getItem('registros')) || [];
// Función para mostrar registros en la tabla de turnos
function mostrarRegistros() {
const tbody = document.getElementById('registrosBody');
tbody.innerHTML = '';
registros.forEach(registro => {
const row = document.createElement('tr');
row.innerHTML = `
<td>${registro.nombreDispatcher}</td>
<td>${registro.horaEntrada}</td>
<td>${registro.horaSalida}</td>
<td>${registro.carrerasPositivas}</td>

<td>${registro.explicacionNegativas || ''}</td>
<td>${registro.carrerasNegativasFaltaDriver}</td>
<td>${registro.carrerasHoteles}</td>
<td>${registro.carrerasWhatsApp}</td>
<td>${registro.appointments}</td>
<td>${registro.observacionesTurno || ''}</td>
`;
tbody.appendChild(row);
});
}
// Función para mostrar drivers en la tabla de drivers
function mostrarDrivers() {
const tbody = document.getElementById('driversBody');
tbody.innerHTML = '';
registros.forEach(registro => {
if (registro.drivers) {
registro.drivers.forEach(driver => {
const row = document.createElement('tr');
row.innerHTML = `
<td>${registro.nombreDispatcher}</td>
<td>${driver.nombre}</td>
<td>${driver.monto}</td>
`;
tbody.appendChild(row);
});
}
});
}
